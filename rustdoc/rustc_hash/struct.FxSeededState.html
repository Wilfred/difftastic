Expand description

[FxSetState] is an alternative state for HashMap types, allowing to use FxHasher with a set seed.

use rustc_hash::FxSeededState;

let mut map = HashMap::with_hasher(FxSeededState::with_seed(12));
map.insert(15, 610);
assert_eq!(map[&15], 610);
