pub const fn ELF_ST_VISIBILITY(o: u8) -> u8 {
    o & 0x03

pub const fn ELF_ST_TYPE(val: u8) -> u8 {
    val & 0xf

    pub r_offset: usize,
    pub r_info: u64,
    pub r_addend: usize,
}

impl Elf_Rela {
    #[inline]
    pub fn type_(&self) -> u32 {
        #[cfg(target_pointer_width = "32")]
        {
            self.r_info & 0xff
        }
        #[cfg(target_pointer_width = "64")]
        {
            (self.r_info & 0xffff_ffff) as u32
        }
    }
}
