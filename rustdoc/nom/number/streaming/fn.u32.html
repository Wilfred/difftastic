Recognizes an unsigned 4 byte integer

If the parameter is nom::number::Endianness::Big, parse a big endian u32 integer, otherwise if nom::number::Endianness::Little parse a little endian u32 integer. Streaming version: Will return Err(nom::Err::Incomplete(_)) if there is not enough data.

use nom::number::streaming::u32;

let be_u32 = |s| {
  u32::<_, (_, ErrorKind)>(nom::number::Endianness::Big)(s)
};

assert_eq!(be_u32(&b"\x00\x03\x05\x07abcefg"[..]), Ok((&b"abcefg"[..], 0x00030507)));
assert_eq!(be_u32(&b"\x01"[..]), Err(Err::Incomplete(Needed::new(3))));

let le_u32 = |s| {
  u32::<_, (_, ErrorKind)>(nom::number::Endianness::Little)(s)
};

assert_eq!(le_u32(&b"\x00\x03\x05\x07abcefg"[..]), Ok((&b"abcefg"[..], 0x07050300)));
assert_eq!(le_u32(&b"\x01"[..]), Err(Err::Incomplete(Needed::new(3))));
