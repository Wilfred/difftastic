Recognizes an unsigned 8 byte integer

If the parameter is nom::number::Endianness::Big, parse a big endian u64 integer, otherwise if nom::number::Endianness::Little parse a little endian u64 integer. Streaming version: Will return Err(nom::Err::Incomplete(_)) if there is not enough data.

use nom::number::streaming::u64;

let be_u64 = |s| {
  u64::<_, (_, ErrorKind)>(nom::number::Endianness::Big)(s)
};

assert_eq!(be_u64(&b"\x00\x01\x02\x03\x04\x05\x06\x07abcefg"[..]), Ok((&b"abcefg"[..], 0x0001020304050607)));
assert_eq!(be_u64(&b"\x01"[..]), Err(Err::Incomplete(Needed::new(7))));

let le_u64 = |s| {
  u64::<_, (_, ErrorKind)>(nom::number::Endianness::Little)(s)
};

assert_eq!(le_u64(&b"\x00\x01\x02\x03\x04\x05\x06\x07abcefg"[..]), Ok((&b"abcefg"[..], 0x0706050403020100)));
assert_eq!(le_u64(&b"\x01"[..]), Err(Err::Incomplete(Needed::new(7))));
