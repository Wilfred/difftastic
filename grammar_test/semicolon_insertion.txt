============================================

  var b = c
  d()

(program
  (if_statement (identifier) (statement_block
    (var_declaration (var_assignment (identifier) (identifier)))
    (expression_statement (function_call (identifier) (arguments)))
    (expression_statement (function_call (identifier) (arguments)))
    (return_statement (identifier)))))
